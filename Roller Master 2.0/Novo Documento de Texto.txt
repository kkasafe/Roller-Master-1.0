<script>
    const resultado = document.getElementById("resultadoBox");
    const som = document.getElementById("somDado");
    const delayBtns = document.querySelectorAll(".delayBtn");
    const rollBtn = document.getElementById("rollBtn");
    const resetBtn = document.getElementById("resetBtn");
    let delay = 5000;

    delayBtns.forEach(btn => {
        btn.addEventListener("click", () => {
            delay = parseInt(btn.dataset.delay);
            delayBtns.forEach(b => b.classList.remove("selected"));
            btn.classList.add("selected");
        });
    });

    document.querySelector('.delayBtn[data-delay="5000"]').classList.add("selected");

    async function rolar() {
        const n = parseInt(document.getElementById("roladas").value);
        const lados = parseInt(document.getElementById("lados").value);

        if (!n || !lados) {
            resultado.innerText = "ERRO";
            return;
        }

        for (let i = 1; i <= n; i++) {

            // Toca som
            som.currentTime = 0;
            som.play();

            const numero = Math.floor(Math.random() * lados) + 1;
            resultado.innerText = numero;

            resultado.style.transform = "scale(1.25)";
            setTimeout(() => resultado.style.transform = "scale(1)", 150);

            if (i < n) await new Promise(r => setTimeout(r, delay));
        }
    }

    function resetar() {
        resultado.innerText = "RESULTADO";
        document.getElementById("roladas").value = "";
        document.getElementById("lados").value = "";
    }

    rollBtn.onclick = rolar;
    resetBtn.onclick = resetar;

    // Enter = rolar
    // Delete = resetar
    document.addEventListener("keydown", e => {
        if (e.key === "Enter") rolar();
        if (e.key === "Delete") resetar();
    });

    document.getElementById("themeToggle").onclick = () =>
        document.body.classList.toggle("dark");

</script>
